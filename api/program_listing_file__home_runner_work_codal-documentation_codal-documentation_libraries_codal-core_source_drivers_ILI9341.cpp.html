namespace codal
{

ILI9341::ILI9341(ScreenIO &io, Pin &cs, Pin &dc) : ST7735(io, cs, dc)
{
    double16 = true;
}

int ILI9341::init()
{
    endCS();
    setData();

    fiber_sleep(10);
    sendCmdSeq(initcmd);

    return DEVICE_OK;
}

} // namespace codal
